\subsubsection{Caso d'uso UC 1.4: Aggiunta di un widget checklist}
\label{UC 1.4: Aggiunta di un widget checklist}
\begin{figure}[ht]
	\centering
	\includegraphics[scale=0.7]{Usecases/img/{UC1.4}.png}
	\caption{Caso d'uso UC 1.4: Aggiunta di un widget checklist}
\end{figure}

\FloatBarrier
\begin{itemize}
\item\textbf{Attori}: Sviluppatore.
\item\textbf{Descrizione}: Lo sviluppatore vuole utilizzare una widget che consiste in una lista formata da elementi ognuno con un campo testo e un bottone di tipo checkbutton.
\item\textbf{Precondizione}: Lo sviluppatore ha accesso all'\termine{SDK}.
\item\textbf{Postcondizione}: Lo sviluppatore generato codice eseguibile per aggiungere un widget checklist a una bolla.
\item\textbf{Scenario principale}:
	\begin{itemize}
		\item Lo sviluppatore vuole personalizzare l'elenco puntato (UC 1.4.1).
		\item Lo sviluppatore vuole impostare un messaggio di completamento (UC1.4.2).
	\end{itemize}
	\item\textbf{Scenario alternativo}:
	\begin{itemize}
		\item Lo sviluppatore immette un messaggio di completamento non valido (UC 1.4.3).
	\end{itemize}
\end{itemize}
